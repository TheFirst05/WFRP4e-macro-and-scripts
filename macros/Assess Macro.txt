// Does not increase advantage over 10. change bypass to false if you want to be able to modify the roll dialog.
const assess = async function (actor) {
    let hasIntuition = game.gmtoolkit.utility.hasSkill(actor, "Intuition", "silent")
    if (hasIntuition != null) {
        var test = await actor.setupSkill(game.i18n.translations.NAME.Intuition, { bypass: true, appendTitle: " to Assess" });
    } else {
        var test = await actor.setupCharacteristic("i", { bypass: true, appendTitle: " to Assess" })
    }
    await test.roll();
    let SL = test.result.SL;
    if (SL > 5) {
        actor.modifyAdvantage(3);
    } else if (test.result.outcome == "success") {
        actor.modifyAdvantage(2);
    };
} 
assess(actor);
